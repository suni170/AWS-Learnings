WITH PTO AS (SELECT T.PersonKey, T.Date
		FROM dbo.FactPersonTime T 
		JOIN dbo.DimTime DT ON T.TimeKey = DT.TimeKey
		WHERE DT.TimeType = 'PTO')
	, MaxProjectForecastDates AS ( 
		SELECT F.ProjectKey
			, CASE WHEN MAX(ISNULL(PT.MaxPTODate,'19000101')) >  MAX(F.MaxDate) THEN MAX(isnull(PT.MaxPTODate,'19000101')) ELSE MAX(F.MaxDate) END [EndDate]
		FROM dbo.FactProjectResourceForecast F 
		JOIN dbo.DimProject P ON F.ProjectKey = P.ProjectKey
		OUTER APPLY (SELECT MAX(PTO.Date) MaxPTODate FROM PTO WHERE PTO.[Date] BETWEEN P.ProjectStartDate AND P.ProjectEndDate AND F.PersonKey = PTO.PersonKey) PT
		WHERE P.ProjectNumber != ''
		GROUP BY F.ProjectKey
	)

	SELECT DISTINCT M.ProjectKey 
	INTO #tmpProjectForecastShortFall
	FROM MaxProjectForecastDates M 
	JOIN dbo.DimProject P ON M.ProjectKey = P.ProjectKey
	WHERE DATEADD(DAY, -2, P.ProjectEndDate) > M.EndDate;
