and (@AdminReviewToDate = '' OR CAST(PD.UCB_AdminReviewDate AS DATE) <= CAST('' + @AdminReviewToDate  + '' AS DATE))

and (@REPSUploadToDate = '' OR CAST(PD.REPSUploadDate AS DATE) <= CAST('' + @REPSUploadToDate  + '' AS DATE))

OR (@SurveillanceReviewRequired = '2' and ISNULL(SurveillanceReviewRequired,0) not in (1) and ISNULL(P.StatusID,0) in (5,103)))                        
                        
        
END                        
-- End of only closed projects                        
IF(@OnlySpecialProjects = 1 OR @IncludeSpecialProjects = 1)                        
BEGIN

        ph.StatusID,
        ph.Modified,

        ph.StatusID IS NOT NULL

PreviousandCurrentStatusFromSpecialProjects_History AS (
    SELECT 
        ProjectID,
        StatusID,
        Modified,
        [Version],
        ROW_NUMBER() OVER (
            PARTITION BY ProjectID 
            ORDER BY Modified DESC, [Version] DESC

    FROM ProjectStatusesFromSpecialProjects_History

SELECT 
    ProjectID,
    StatusID,
    Modified
INTO #Temp_MaxModified_SpecialProjectHistory

WHERE RowNum = 1;

  select DISTINCT P.ProjectID,                           
  0 as [ParentProjectID],                        
    ISNULL(P.MediciNo,'') as[MediciNo],                                                                                                                           
    ISNULL(P.CBWOrderNo,'') as[CBWOrderNo],                                                       
    isnull(PT.Description,'') as [ProjectType],                                                                                                                                           
   (SELECT Isnull (((SELECT Substring(Replace(Replace(tbl.ids, '<T>', ''), '</T>', ','), 0,Len(Replace(Replace(tbl.ids, '<T>', ''),'</T>', ',')))                                                           
             FROM   (SELECT T . t                                                           
                                            FROM   (SELECT Description                                                           
                                                    FROM   mststandards WITH(NOLOCK)

     FOR xml path('')) AS tbl(ids))), ''))  AS [STDDescription],                         
    ISNULL(P.StatusID, 0) AS [StatusID],                          
    isnull(Status,'') as [Status],                                                                                          
    isnull(P.ClientNumber,'') as [ClientNumber],                                                                      
    isnull(P.CompanyName,'') as [CompanyName],                                                                                                                      
    ISNULL(Reg.RegionName,'') as [RegionName],                                                                                           
    isnull(DepartmentName,'') as [Department],                                                                                                                              
    isnull(PHUSR.UserName,'') as [ProjectHandler],                                    
    ISNULL(CONVERT(varchar(10),P.CertificateExpirationDate,126),'')  AS [CertificateExpirationDate],                                                                                                             
'' as [TC1],                                                                                                                                                                                                                            
    COALESCE(nullif(STC3.UserName,'') ,nullif(STC2.UserName,''),nullif(STC1.UserName,'')) as [TC2],                                
    ISNULL(P.ApplicableCB,'') as [ApplicableCB],                                            
    ISNULL(P.Indicator, '') AS [ProductArea],                                                                                                           
    CAST(1 as BIT) AS [IsSpecial],                                  
      ISNULL(ABS(DATEDIFF(dd, P.DateofSubmission, getdate())),'') as [DaysInMedici],                            
    ISNULL(ABS(DATEDIFF(dd, (PH.Modified), getdate())),'') as DaysInCurrentStatus,                            
   ISNULL(CONVERT(varchar(10),P.DateofSubmission,126),'') as [DateofSubmission],                                                                                                                          
   ISNULL(CONVERT(varchar(10),P.LastAuditDay,126),'') as [LastAuditDay],                                                                                         
   ISNULL(P.CertificateNumber, '') AS [CertificateNumber],                                                                                                  
   ''  as [MDCodes],                                                                                      
   ISNULL(DisappearanceDate, '') as DisappearanceDate,                                                                      
    CASE                                                                              
        WHEN DATEDIFF(day, Getdate(), ISNULL(MSD.Deadline, Getdate())) > 0 THEN + CAST(DATEDIFF(day, Getdate(), ISNULL(MSD.Deadline, Getdate())) AS VARCHAR(10))                                                                              
        WHEN DATEDIFF(day, Getdate(), ISNULL(MSD.Deadline, Getdate())) < 0 THEN CAST(DATEDIFF(day, Getdate(), ISNULL(MSD.Deadline, Getdate())) AS VARCHAR(10))                                                                              
        ELSE ''                                                    
    END AS [Deadline],                        
 '' AS [MHSAR],              
 '' as [TechnicalReviewer],                         
 '' AS AdminReviewer,

 '' AS [CertificateReviewer]  ,                
   ISNULL(GLAUSR.UserName,'') AS GLAManager

 and ((ISNULL(@IncludeClosedProjects, 0) = 1 OR ISNULL(@OnlyClosedProjects, 0) = 1 ) OR P.StatusID NOT in (12,13,111,24,25,26,27,28,29,51,52,53,54,55,56,57))                             
 and (@StatusID = '' OR P.StatusID in (select value from FunclisttotableInt(@StatusID, ',')))                        
 and (@CertExpiryToDt = '' OR P.CertificateExpirationDate IS NOT NULL and  CAST(P.CertificateExpirationDate AS DATE) <= CAST('' + @CertExpiryToDt + '' AS DATE))                       
 and (@CertExpiryToDt = '' OR P.CertificateExpirationDate IS NOT NULL and  CAST(P.CertificateExpirationDate AS DATE) <= CAST('' + @CertExpiryToDt + '' AS DATE))                             
 and (@RegistrationFromDt = '' OR CAST(P.DateOfSubmission AS DATE) >= CAST('' + @RegistrationFromDt + '' AS DATE))

 and (@STCLoginID = '' OR COALESCE(nullif(EVL.STCFinalEval,'') ,nullif(EVL.STCSecondEval,''),nullif(EVL.STC,'')) = @STCLoginID)
